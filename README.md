## 1. ü´ß –ü—É–∑—ã—Ä—å–∫–æ–≤–∞—è —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ (Bubble Sort)

            // –ï—Å–ª–∏ —Ç–µ–∫—É—â–∏–π —ç–ª–µ–º–µ–Ω—Ç –º–µ–Ω—å—à–µ –ø—Ä–µ–¥—ã–¥—É—â–µ–≥–æ (–Ω–∞—Ä—É—à–µ–Ω –ø–æ—Ä—è–¥–æ–∫)

int main() {
    // –ú–∞—Å—Å–∏–≤ —Ç–∏–ø–∞ short (—ç–∫–æ–Ω–æ–º–∏–º –ø–∞–º—è—Ç—å –Ω–∞ –¥–∞–Ω–Ω—ã—Ö)

        // –ü—Ä–µ–¥—Å—Ç–∞–≤–ª—è–µ–º, —á—Ç–æ —Ç–µ–∫—É—â–∏–π —ç–ª–µ–º–µ–Ω—Ç i - —Å–∞–º—ã–π –º–∞–ª–µ–Ω—å–∫–∏–π

        // –ò—â–µ–º —Ä–µ–∞–ª—å–Ω—ã–π –º–∏–Ω–∏–º—É–º –≤ –æ—Å—Ç–∞–≤—à–µ–π—Å—è —á–∞—Å—Ç–∏ –º–∞—Å—Å–∏–≤–∞

            a[minIndex] = box;
        }

int main() {
    // –°–æ–∑–¥–∞–µ–º –¥–∏–Ω–∞–º–∏—á–µ—Å–∫–∏–π –º–∞—Å—Å–∏–≤

    // –í–≤–æ–¥
    for (short i = 0; i < 7; i++) {
        cin >> a[i];
    }

    // --- –°–û–†–¢–ò–†–û–í–ö–ê ---
    // –ù–∞—á–∏–Ω–∞–µ–º —Å–æ 2-–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞ (–∏–Ω–¥–µ–∫—Å 1), —Ç–∞–∫ –∫–∞–∫ 1-–π —Å—á–∏—Ç–∞–µ—Ç—Å—è –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã–º

        short key = a[i];  // –ó–∞–ø–æ–º–∏–Ω–∞–µ–º –∫–∞—Ä—Ç—É, –∫–æ—Ç–æ—Ä—É—é —Ö–æ—Ç–∏–º –≤—Å—Ç–∞–≤–∏—Ç—å

            a[j + 1] = a[j]; // –°–¥–≤–∏–≥–∞–µ–º —Å–æ—Å–µ–¥–∞ –≤–ø—Ä–∞–≤–æ
